\begin{frame}[fragile]
    \frametitle{Binary Search}
    \lstset{language=C,
            basicstyle=\footnotesize\ttfamily,
            numbers=left,
            numberstyle=\tiny\color{gray},
            numbersep=5pt,
            showspaces=false,
            showstringspaces=false,
            breaklines=true,
            breakatwhitespace=false,
            keywordstyle=\color{blue},
            commentstyle=\color{OliveGreen},
            stringstyle=\color{mauve},
            morekeywords={*,...},
            moredelim=**[is][\btHL]{`}{`}}
    \begin{lstlisting}
    int binary_search(int *array, int n, int x)
    {
        int lo = 0, hi = n - 1;
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            if (array[mid] < x)
                lo = mid + 1;
            else hi = mid;
        }

        if (lo == hi && array[lo] == x)
            return lo;
        return -1;
    }
    \end{lstlisting}
\end{frame}
